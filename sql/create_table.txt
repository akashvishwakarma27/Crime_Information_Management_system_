  CREATE TABLE POLICE_STATION(
  PS_ID Integer PRIMARY KEY AUTO_INCREMENT,
  PS_NAME VARCHAR(15) UNIQUE
  );
  
  CREATE TABLE VICTIM(
  VICTIM_ID INTEGER PRIMARY KEY AUTO_INCREMENT,
  VICTIM_NAME VARCHAR(15),
  VICTIM_AGE INTEGER,
  VICTIM_GENDER VARCHAR(10)
  );
  
  CREATE TABLE AREA(
  AREA_ID INTEGER PRIMARY KEY AUTO_INCREMENT,
  AREA_NAME VARCHAR(15) UNIQUE
  );
  
  CREATE TABLE CRIMINAL(
  CRIMINAL_ID INTEGER PRIMARY KEY AUTO_INCREMENT,
  CRIMINAL_NAME VARCHAR(15),
  CRIMINAL_AGE INTEGER,
  CRIMINAL_GENDER VARCHAR(10),
  CRIMINAL_ADDRESS VARCHAR(15),
  CRIMINAL_IDENTIFICATION_MARK VARCHAR(20)
  );
  
  CREATE TABLE CRIME(
  CRIME_ID INTEGER PRIMARY KEY AUTO_INCREMENT,
  CRIME_DATE DATE,
  CRIME_TYPE VARCHAR(20),
  CRIME_DETAILS VARCHAR(100),
  CRIME_STATUS VARCHAR(15),
  CRIME_SUSPECT VARCHAR(100),
  CRIMINAL_ID INT,
  AREA_ID INT,
  PS_ID INT,
  VICTIM_ID INT,
  FOREIGN KEY(criminal_id) REFERENCES CRIMINAL(criminal_id),
  FOREIGN KEY(area_id) REFERENCES AREA(area_id),
  FOREIGN KEY(ps_id) REFERENCES POLICE_STATION(ps_id),
  FOREIGN KEY(victim_id) REFERENCES VICTIM(victim_id)
  );
